    <script>
        const apiKey = ""; 

        async function callGemini(prompt, systemInstruction) {
            const payload = {
                contents: [{ parts: [{ text: prompt }] }],
                systemInstruction: { parts: [{ text: systemInstruction }] }
            };
            for (let i = 0; i < 5; i++) {
                try {
                    const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    });
                    const data = await response.json();
                    return data.candidates?.[0]?.content?.parts?.[0]?.text;
                } catch (e) {
                    await new Promise(resolve => setTimeout(resolve, Math.pow(2, i) * 1000));
                }
            }
            throw new Error("Erro de ligação.");
        }

        async function generateConcept() {
            const input = document.getElementById('style-input').value;
            if (!input) return;
            const btn = document.getElementById('concept-btn');
            const resultDiv = document.getElementById('concept-result');
            btn.disabled = true;
            btn.innerHTML = `<span class="loader"></span> A criar...`;
            resultDiv.classList.add('hidden');
            try {
                const prompt = `Sugere um conceito fotográfico profissional para: ${input}. Responde em PT-PT.`;
                const system = "És um consultor criativo de fotografia de luxo.";
                const text = await callGemini(prompt, system);
                resultDiv.innerHTML = text.replace(/\n/g, '<br>');
                resultDiv.classList.remove('hidden');
            } catch (e) {
                resultDiv.innerHTML = "Erro ao gerar ideias.";
                resultDiv.classList.remove('hidden');
            } finally {
                btn.disabled = false;
                btn.innerText = "Gerar Ideias Criativas";
            }
        }

        async function generateQuote() {
            const input = document.getElementById('quote-input').value;
            if (!input) return;
            const btn = document.getElementById('quote-btn');
            const resultDiv = document.getElementById('quote-result');
            btn.disabled = true;
            btn.innerHTML = `<span class="loader"></span> A calcular...`;
            resultDiv.classList.add('hidden');
            try {
                const prompt = `Cria uma proposta amigável para este evento: ${input}. Responde em PT-PT.`;
                const system = "És o gestor comercial da Casttiel Fotografias.";
                const text = await callGemini(prompt, system);
                resultDiv.innerHTML = text.replace(/\n/g, '<br>');
                resultDiv.classList.remove('hidden');
            } catch (e) {
                resultDiv.innerHTML = "Erro ao calcular orçamento.";
                resultDiv.classList.remove('hidden');
            } finally {
                btn.disabled = false;
                btn.innerText = "Pedir Proposta Rápida";
            }
        }
    </script>
